# Internationalization (i18n)

## Philosophy

Every user-facing string goes through localization from day one. No hardcoded strings in components. react-i18next provides the translation engine. JSON files per language, separated by namespace (feature). The language preference is persisted in Zustand and survives page reloads. Date and number formatting use the browser's built-in `Intl` API -- not a library.

## i18n Configuration

```typescript
// lib/i18n.ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import LanguageDetector from 'i18next-browser-languagedetector';

// Import translation files
import enCommon from '@/locales/en/common.json';
import enAuth from '@/locales/en/auth.json';
import enDashboard from '@/locales/en/dashboard.json';
import enUsers from '@/locales/en/users.json';

import trCommon from '@/locales/tr/common.json';
import trAuth from '@/locales/tr/auth.json';
import trDashboard from '@/locales/tr/dashboard.json';
import trUsers from '@/locales/tr/users.json';

export const supportedLanguages = ['en', 'tr'] as const;
export type SupportedLanguage = (typeof supportedLanguages)[number];

i18n
  .use(LanguageDetector)
  .use(initReactI18next)
  .init({
    resources: {
      en: {
        common: enCommon,
        auth: enAuth,
        dashboard: enDashboard,
        users: enUsers,
      },
      tr: {
        common: trCommon,
        auth: trAuth,
        dashboard: trDashboard,
        users: trUsers,
      },
    },
    defaultNS: 'common',
    fallbackLng: 'en',
    interpolation: {
      escapeValue: false, // React already escapes
    },
    detection: {
      order: ['localStorage', 'navigator'],
      lookupLocalStorage: 'language',
      caches: ['localStorage'],
    },
  });

export default i18n;
```

## Translation File Structure

```
src/
  locales/

      auth.json          # Login, register, forgot password
      dashboard.json     # Dashboard-specific
      users.json         # User management

      auth.json
      dashboard.json
      users.json
```

## useTranslation Hook Usage

```typescript
// features/auth/login-page.tsx
import { useTranslation } from 'react-i18next';

export function LoginPage() {
  const { t } = useTranslation('auth'); // namespace = 'auth'

  return (
    <div>
      <h1>{t('login.title')}</h1>
      <form>
        <label>{t('login.email')}</label>
        <input type="email" />

        <label>{t('login.password')}</label>
        <input type="password" />

        <button type="submit">{t('login.submit')}</button>

        <a href="/forgot-password">{t('login.forgotPassword')}</a>
      </form>
    </div>
  );
}
```

## Cross-Namespace Access

```typescript
// Access common namespace from any component
const { t } = useTranslation('users');

// Use common namespace keys with prefix
<button>{t('common:actions.save')}</button>
<button>{t('common:actions.cancel')}</button>

// Or get multiple namespaces
const { t } = useTranslation(['users', 'common']);
<p>{t('common:errors.generic')}</p>
```

## Plurals

```typescript
const { t } = useTranslation();

// common.json: "items_one": "{{count}} item", "items_other": "{{count}} items"
t('items', { count: 1 });  // "1 item"
t('items', { count: 5 });  // "5 items"
t('items', { count: 0 });  // "0 items"
```

## Language Switcher Component

```typescript
// components/ui/language-switcher.tsx
import { useTranslation } from 'react-i18next';
import { supportedLanguages, SupportedLanguage } from '@/lib/i18n';

const languageLabels: Record<SupportedLanguage, string> = {
  en: 'English',
  tr: 'Turkce',
};

const languageFlags: Record<SupportedLanguage, string> = {
  en: 'GB',
  tr: 'TR',
};

export function LanguageSwitcher() {
  const { i18n } = useTranslation();
  const currentLang = i18n.language as SupportedLanguage;

  const handleChange = (lang: SupportedLanguage) => {
    i18n.changeLanguage(lang);
    // i18next-browser-languagedetector saves to localStorage automatically
  };

  return (
    <div className="flex items-center gap-1 rounded-md border p-1">
      {supportedLanguages.map((lang) => (
        <button
          key={lang}
          onClick={() => handleChange(lang)}
          className={`rounded px-2 py-1 text-xs font-medium transition-colors ${
            currentLang === lang
              ? 'bg-blue-100 text-blue-700'
              : 'text-gray-500 hover:text-gray-700'
          }`}
        >
          {languageLabels[lang]}
        </button>
      ))}
    </div>
  );
}
```

## Date and Number Formatting with Intl

No library needed. The browser's `Intl` API handles locale-aware formatting.

```typescript
// lib/formatters.ts
import i18n from '@/lib/i18n';

export function formatDate(date: string | Date, style: 'short' | 'medium' | 'long' = 'medium'): string {
  const d = typeof date === 'string' ? new Date(date) : date;
  const locale = i18n.language;

  const options: Record<string, Intl.DateTimeFormatOptions> = {
    short: { month: 'short', day: 'numeric' },
    medium: { year: 'numeric', month: 'short', day: 'numeric' },
    long: { year: 'numeric', month: 'long', day: 'numeric', hour: '2-digit', minute: '2-digit' },
  };

  return new Intl.DateTimeFormat(locale, options[style]).format(d);
}

export function formatNumber(value: number): string {
  return new Intl.NumberFormat(i18n.language).format(value);
}

export function formatCurrency(value: number, currency = 'USD'): string {
  return new Intl.NumberFormat(i18n.language, {
    style: 'currency',
    currency,
  }).format(value);
}

export function formatPercent(value: number): string {
  return new Intl.NumberFormat(i18n.language, {
    style: 'percent',
    minimumFractionDigits: 1,
  }).format(value);
}

export function formatRelativeTime(date: string | Date): string {
  const d = typeof date === 'string' ? new Date(date) : date;
  const now = new Date();
  const diffMs = now.getTime() - d.getTime();
  const diffSeconds = Math.floor(diffMs / 1000);
  const diffMinutes = Math.floor(diffSeconds / 60);
  const diffHours = Math.floor(diffMinutes / 60);
  const diffDays = Math.floor(diffHours / 24);

  const rtf = new Intl.RelativeTimeFormat(i18n.language, { numeric: 'auto' });

  if (diffSeconds < 60) return rtf.format(-diffSeconds, 'second');
  if (diffMinutes < 60) return rtf.format(-diffMinutes, 'minute');
  if (diffHours < 24) return rtf.format(-diffHours, 'hour');
  if (diffDays < 30) return rtf.format(-diffDays, 'day');

  return formatDate(d, 'medium');
}
```

## Key Rules

1. **No hardcoded strings** -- every user-visible string uses `t()`. Labels, buttons, errors, placeholders, tooltips -- everything.
2. **Namespace per feature** -- `common` for shared, `auth` for login/register, `dashboard` for dashboard. One JSON file per namespace per language.
3. **English is fallback** -- `fallbackLng: 'en'`. If a Turkish key is missing, the English version shows.
4. **Intl for dates/numbers** -- never format dates or numbers manually. Use `Intl.DateTimeFormat` and `Intl.NumberFormat` with the current locale.
5. **Plurals use i18next convention** -- `_one` and `_other` suffixes. Pass `{ count }` to the `t()` function.
6. **Language persisted in localStorage** -- `i18next-browser-languagedetector` auto-detects and saves. No Zustand needed for language persistence.
7. **Import i18n before React** -- `import '@/lib/i18n'` at the top of `main.tsx`, before any component renders.
8. **Flat keys with dots** -- `t('login.title')` maps to `{ "login": { "title": "..." } }`. Keep nesting to 2-3 levels max.
9. **Cross-namespace with prefix** -- `t('common:actions.save')` when accessing common translations from a feature namespace.
10. **Add both languages at once** -- when adding a new key, add it to both `en` and `tr` JSON files in the same commit. Never leave a language behind.
